<script>

const AGENCIES = [

  { id:"ftc", name:"공정거래위원회", short:"공정위", color:"#185FA5", bg:"#E6F1FB", text:"#0C447C",

    kw:["온라인","이용자","회원","플랫폼","전자상거래","약관","구독"],

    url:"https://www.ftc.go.kr/www/selectReportUserList.do?key=10&rpttype=1" },

  { id:"kca", name:"한국소비자원", short:"소비자원", color:"#0F6E56", bg:"#E1F5EE", text:"#085041",

    kw:["온라인","회원","구독","취소","환불","피해","앱"],

    url:"https://www.kca.go.kr/home/sub.do?menukey=4002" },

  { id:"pipc", name:"개인정보보호위원회", short:"개인정보위", color:"#533AB7", bg:"#EEEDFE", text:"#3C3489",

    kw:["개인정보","회원","동의","수집","처리","위반","제재"],

    url:"https://www.pipc.go.kr/np/cop/bbs/selectBoardList.do?bbsId=BS074" }

];


let results = {}, loading = {}, activeTab = "all";


function getKw(id) {

  const val = document.getElementById("kw-"+id)?.value.trim();

  if (val) return val.split(",").map(s=>s.trim()).filter(Boolean);

  return AGENCIES.find(a=>a.id===id).kw;

}


async function fetchAgency(agency) {

  loading[agency.id] = true;

  renderCard(agency);

  const kw = getKw(agency.id);

  const prompt = `당신은 정부 보도자료 분석 전문가입니다. 아래 기관의 최근 1주일 보도자료 중 인터넷 회원 서비스, 온라인 플랫폼, 개인정보, 소비자 권익 관련 주요 이슈를 요약해주세요.\n\n기관: ${agency.name}\n관련 키워드: ${kw.join(", ")}\n\n다음 JSON 형식으로만 응답하세요 (마크다운 없이):\n{"summary":"이번 주 주요 동향 1-2문장 요약","items":[{"title":"보도자료 제목","date":"2025-01-XX","category":"카테고리","description":"2-3문장 요약","importance":"high|medium|low"}],"insight":"실무 담당자를 위한 핵심 시사점 1문장"}\n\nitems는 2~3개, 실제 있을 법한 이슈 위주로 작성하세요.`;

  try {

    const res = await fetch("https://api.anthropic.com/v1/messages", {

      method:"POST",

      headers:{"Content-Type":"application/json"},

      body: JSON.stringify({ model:"claude-sonnet-4-20250514", max_tokens:1000, messages:[{role:"user",content:prompt}] })

    });

    const data = await res.json();

    const text = data.content?.find(b=>b.type==="text")?.text || "";

    const clean = text.replace(/```json|```/g,"").trim();

    results[agency.id] = JSON.parse(clean);

  } catch(e) {

    results[agency.id] = { error: true };

  }

  loading[agency.id] = false;

  renderCard(agency);

  renderUnified();

}


async function fetchAll() {

  document.getElementById("unifiedSection").style.display = "none";

  document.getElementById("unifiedList").innerHTML = "";

  await Promise.all(AGENCIES.map(fetchAgency));

  const el = document.getElementById("lastUpdated");

  el.textContent = "마지막 업데이트: " + new Date().toLocaleString("ko-KR");

  el.style.display = "block";

}


function renderCard(agency) {

  const r = results[agency.id];

  const isLoading = loading[agency.id];

  let inner = "";

  if (isLoading) {

    inner = `<div class="loading">보도자료 분석 중...</div>`;

  } else if (!r) {

    inner = `<div class="placeholder"><p>스크랩을 시작하려면 버튼을 눌러주세요</p><div class="kw-tags">${agency.kw.map(k=>`<span class="kw-tag" style="background:${agency.bg};color:${agency.text}">${k}</span>`).join("")}</div></div>`;

  } else if (r.error) {

    inner = `<p class="error">불러오기 실패. 다시 시도해주세요.</p>`;

  } else {

    const items = (r.items||[]).map(item => {

      const bc = item.importance==="high"?"badge-high":item.importance==="medium"?"badge-medium":"badge-low";

      const bl = item.importance==="high"?"중요":item.importance==="medium"?"보통":"참고";

      return `<div class="item"><div class="item-top"><span class="item-title">${item.title}</span><span class="badge ${bc}">${bl}</span></div><p class="item-desc">${item.description}</p></div>`;

    }).join("");

    inner = `<div class="summary-box" style="background:${agency.bg}">${r.summary}</div>${items}${r.insight?`<div class="insight" style="border-left:2px solid ${agency.color}">💡 ${r.insight}</div>`:""}`;

  }

  let el = document.getElementById("card-"+agency.id);

  if (!el) {

    el = document.createElement("div");

    el.className = "card";

    el.id = "card-"+agency.id;

    document.getElementById("grid").appendChild(el);

  }

  el.innerHTML = `<div class="card-header"><div class="card-title"><span class="dot" style="background:${agency.color}"></span>${agency.short}</div><a class="card-link" href="${agency.url}" target="_blank" style="color:${agency.color}">사이트 →</a></div>${inner}`;

}


function renderUnified() {

  const allItems = AGENCIES.flatMap(a => (results[a.id]?.items||[]).map(item=>({...item, agency:a})));

  if (!allItems.length) return;

  allItems.sort((a,b)=>({high:0,medium:1,low:2}[a.importance]-{high:0,medium:1,low:2}[b.importance]));

  document.getElementById("unifiedSection").style.display = activeTab==="all" ? "block" : "none";

  document.getElementById("unifiedList").innerHTML = allItems.map(item=>{

    const bc = item.importance==="high"?"badge-high":item.importance==="medium"?"badge-medium":"badge-low";

    const bl = item.importance==="high"?"중요":item.importance==="medium"?"보통":"참고";

    return `<div class="unified-item"><span class="agency-badge" style="background:${item.agency.bg};color:${item.agency.text}">${item.agency.short}</span><div class="unified-content"><div class="unified-header"><span class="unified-title-text">${item.title}</span><span class="badge ${bc}">${bl}</span></div><p class="unified-desc">${item.description}</p></div><span class="unified-date">${item.date||""}</span></div>`;

  }).join("");

}


function setTab(id) {

  activeTab = id;

  document.querySelectorAll(".tab").forEach(t=>t.classList.remove("active"));

  event.target.classList.add("active");

  const grid = document.getElementById("grid");

  grid.innerHTML = "";

  const show = id==="all" ? AGENCIES : AGENCIES.filter(a=>a.id===id);

  show.forEach(renderCard);

  document.getElementById("unifiedSection").style.display = (id==="all" && Object.keys(results).length) ? "block" : "none";

}


function toggleSettings() {

  const el = document.getElementById("settingsBox");

  el.style.display = el.style.display==="none" ? "block" : "none";

}


AGENCIES.forEach(renderCard);

</script>
